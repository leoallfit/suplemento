Precios desde Google Sheets

Sin configurar

Gestioná los precios desde una planilla. Cada vez que alguien abre la página los precios se actualizan automáticamente desde tu Google Sheet.

1
Creá la planilla en Google Sheets
Copiá la plantilla con tus productos y pegala en una hoja nueva. Las columnas son: ID · Nombre · Precio Venta · Precio Lista · Ganancia $ · Ganancia % (las últimas dos son fórmulas automáticas).
2
Creá un Apps Script en tu planilla
En Google Sheets → Extensiones → Apps Script. Borrá todo el código y pegá este:
function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  // ── CRM: registrar pedido ─────────────────────────────
  if (e && e.parameter && e.parameter.action === 'newOrder') {
    var p = ss.getSheetByName('Pedidos');
    if (!p) {
      p = ss.insertSheet('Pedidos');
      p.appendRow(['# Pedido','Fecha','Cliente','Hora','Productos','Total',
                   'Estado','Teléfono','Notas']);
      p.getRange(1,1,1,9).setFontWeight('bold')
        .setBackground('#1c4587').setFontColor('#ffffff');
      p.setFrozenRows(1); p.setColumnWidth(5,350);
    }
    var now = new Date(), tz = Session.getScriptTimeZone();
    var num = p.getLastRow();
    p.appendRow(['#'+num,
      Utilities.formatDate(now,tz,'dd/MM/yyyy'),
      e.parameter.cliente||'',
      Utilities.formatDate(now,tz,'HH:mm'),
      e.parameter.items||'', e.parameter.total||'',
      'Pendiente','','']);
    return ContentService
      .createTextOutput(JSON.stringify({ok:true,orderId:'#'+num}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // ── Productos: agregar/actualizar ─────────────────────
  if (e && e.parameter && e.parameter.action === 'addProduct') {
    var s = ss.getSheets()[0];
    var id = e.parameter.id, name = e.parameter.name;
    var price = parseFloat(e.parameter.sale_price) || 0;
    var d = s.getDataRange().getValues(), rowIdx = -1;
    for (var i = 1; i < d.length; i++) {
      if (String(d[i][0]) === String(id)) { rowIdx = i+1; break; }
    }
    if (rowIdx > 0) {
      s.getRange(rowIdx,2).setValue(name);
      s.getRange(rowIdx,3).setValue(price);
      s.getRange(rowIdx,5).setFormula('=C'+rowIdx+'-D'+rowIdx);
      s.getRange(rowIdx,6).setFormula('=IF(D'+rowIdx+'>0,((C'+rowIdx+'-D'+rowIdx+')/D'+rowIdx+')*100,"")');
    } else {
      var row = d.length+1;
      s.getRange(row,1).setValue(id);
      s.getRange(row,2).setValue(name);
      s.getRange(row,3).setValue(price);
      s.getRange(row,5).setFormula('=C'+row+'-D'+row);
      s.getRange(row,6).setFormula('=IF(D'+row+'>0,((C'+row+'-D'+row+')/D'+row+')*100,"")');
    }
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // ── Lista de precios ──────────────────────────────────
  var s = ss.getSheets()[0];
  var d = s.getDataRange().getValues(), rows = [];
  for (var i = 1; i < d.length; i++) {
    if (!d[i][0]) continue;
    rows.push({id:d[i][0], name:d[i][1],
               sale_price:d[i][2], list_price:d[i][3]});
  }
  return ContentService
    .createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}
Después: Implementar → Administrar implementaciones → ✏️ Editar → Nueva versión → Implementar. La URL no cambia. Si es la primera vez: Nueva implementación → Aplicación web → Acceso: Cualquiera → Implementar.
✓ Al agregar un producto nuevo se añade a la planilla (sin tocar Precio Lista).
✓ Cada pedido de WhatsApp se registra automáticamente en la pestaña Pedidos con fecha, hora, productos y total. Completás Cliente y Teléfono vos mismo.
